Usage

import { FormField, Input, Textarea, Select, Checkbox, Radio, RadioGroup } from './Form';

// Simple input with label + helper
<FormField label="Email" helper="Para enviar o link" required>
  <Input type="email" name="email" />
</FormField>

// Input with error (auto-wires aria-invalid + aria-describedby)
<FormField label="Email" error="Email inválido">
  <Input type="email" defaultValue="abc" />
</FormField>

// Input with icon slots
<FormField label="Buscar">
  <Input
    type="search"
    leftSlot={<SearchIcon />}
    placeholder="Digite para buscar..."
  />
</FormField>

// Currency input (right slot)
<FormField label="Valor">
  <Input type="number" rightSlot="BRL" />
</FormField>

// Textarea with character count
<FormField label="Mensagem" helper="Min 50 chars">
  <Textarea rows={4} maxLength={500} showCount />
</FormField>

// Auto-grow textarea
<FormField label="Mensagem">
  <Textarea autoGrow rows={2} />
</FormField>

// Native select
<FormField label="Cargo">
  <Select
    placeholder="Selecione"
    options={[
      { value: 'ceo', label: 'CEO / Founder' },
      { value: 'vp', label: 'VP / C-Level' },
      { value: 'manager', label: 'Manager' }
    ]}
  />
</FormField>

// Or with children
<FormField label="Plano">
  <Select>
    <option value="">Selecione</option>
    <option value="pro">Pro</option>
    <option value="business">Business</option>
  </Select>
</FormField>

// Standalone checkbox (own label, no FormField needed)
<Checkbox
  label="Aceito os termos"
  helper="Política de privacidade"
  required
/>

// Indeterminate checkbox (mixed state)
<Checkbox label="Selecionar todos" indeterminate />

// Radio group
<RadioGroup label="Tamanho da empresa" name="size" required>
  <Radio value="pme" label="PME" />
  <Radio value="mid" label="Mid-market" />
  <Radio value="ent" label="Enterprise" />
</RadioGroup>
