Usage

// Standard section header (most common)
<SectionHeader
  eyebrow="O método"
  title={<>Não é um curso. É uma <em>travessia</em>.</>}
  description="A IA não é tendência..."
/>

// Split layout (heading + lede in 2 cols)
<SectionHeader
  layout="split"
  eyebrow="A jornada"
  title={<>Cinco encontros. Uma <em>nova operação</em>.</>}
  description="Cada encontro é uma estação..."
/>

// Centered (FAQ header, intros)
<SectionHeader
  align="center"
  eyebrow="Antes de decidir"
  title={<>Perguntas <em>frequentes</em></>}
/>

// Dark closing CTA
<SectionHeader
  align="center"
  tone="dark"
  eyebrowTone="accent"
  eyebrow="Próximo passo"
  title={<>Pronto para <em>orquestrar</em>?</>}
  description="O diagnóstico é gratuito."
/>

// With CTA action
<SectionHeader
  layout="split"
  eyebrow="Investimento"
  title="Uma decisão"
  description="Programa completo..."
  actions={<Button variant="primary" arrow>Agendar</Button>}
/>

// Without dash (sub-header style)
<SectionHeader
  withDash={false}
  eyebrow="Mentoria executiva · 2026"
  title="Para líderes que conduzem mudança"
/>

// Custom heading level (h3 instead of h2)
<SectionHeader
  level={3}
  eyebrow="Subseção"
  title="Algo aninhado"
/>

// With aria-labelledby (semantic section)
<section aria-labelledby="metodo">
  <SectionHeader
    id="metodo"  // generates id="metodo-title"
    eyebrow="O método"
    title="Não é um curso..."
  />
  ...
</section>
