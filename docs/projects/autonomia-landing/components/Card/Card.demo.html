Usage

// Pilar (era .a-pillar)
<Card variant="numbered"
      eyebrow="01 · Tecnologia"
      title={<>Domínio <em>aplicado</em></>}
      body="IA como ferramenta operacional..." />

// Módulo da jornada (era .a-module)
<Card variant="numbered"
      eyebrow="01"
      title="Diagnóstico estratégico"
      body="Antes do compromisso..."
      meta="60 min · gratuito" />

// Stat (era .a-stat)
<Card variant="stat"
      stat={<>100<em>%</em></>}
      body="de garantia se os resultados..."
      source="Compromisso" />

// Feature (era .a-diag-feature)
<Card variant="feature"
      icon={<CheckIcon />}
      title="Mapeamento estratégico"
      body="Estado atual do negócio..." />

// Elevated card (Diagnóstico hero)
<Card tone="elevated" accentBar
      eyebrow="Passo zero · gratuito"
      title={<>Comece pelo <em>diagnóstico</em></>}
      body="60 minutos...">
  <Button variant="primary" arrow>Agendar agora</Button>
</Card>

// Interactive card (clickable)
<Card variant="numbered" tone="elevated" interactive
      href="/metodo"
      eyebrow="Método"
      title="Como funciona"
      body="Os 3 pilares..." />
